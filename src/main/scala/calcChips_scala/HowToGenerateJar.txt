1. setup environment
export JAVA_HOME="/c/Program Files/Java/jdk-11.0.0.2"
export PATH=$JAVA_HOME/bin:$PATH

2. check java version
echo $JAVA_HOME
java -version
javac -version

3. build jar
mvn clean package assembly:single

4. check jar
jar tf target/calcChips_arg-jar-with-dependencies.jar
